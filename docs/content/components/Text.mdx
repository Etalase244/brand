import {TextSizes} from '@primer/react-brand'
import {Box as Container} from '@primer/react'

```js
import {Text} from '@primer/react-brand'
```

## Examples

### Default

```jsx live
<Container pl={3}>
  <Text>With GitHub Copilot, you’re always in charge.</Text>
</Container>
```

### Scale

```jsx live
<Container pl={3}>
  {TextSizes.map((size) => (
    <Text key={size} size={size} as="p">
      {size}. With GitHub Copilot, you’re always in charge.
    </Text>
  ))}
</Container>
```

### Polymorphism

The `Text` component can render as a `span`, `p` or `div` HTML element.

```jsx live
<Container pl={3}>
  <Text as="p">With GitHub Copilot, you’re always in charge.</Text>
</Container>
```

## Component props

### Text

| Name   | Type                                                                                   | Default  | Description                         |
| :----- | :------------------------------------------------------------------------------------- | :------: | :---------------------------------- |
| `as`   | `'p'`<br /> `'span'`<br /> `'div'`                                                     | `'span'` | Applies the underlying HTML element |
| `size` | `'700'`<br />`'600'`<br />`'500'`<br />`'400'`<br />`'300'`<br />`'200'`<br />`'100' ` | `'400'`  | Specify the text size               |
