import {
  TextSizes,
  TextTags,
  defaultTextTag,
  defaultTextSize,
} from '@primer/react-brand'
import {Box as Container} from '@primer/react'
import {PropTableValues} from '../../src/components'

```js
import {Text} from '@primer/react-brand'
```

## Examples

### Default

```jsx live
<Container pl={3}>
  <Text>With GitHub Copilot, you’re always in charge.</Text>
</Container>
```

### Scale

```jsx live
<Container pl={3}>
  {TextSizes.map((size) => (
    <Text key={size} size={size} as="p">
      Text {size}
    </Text>
  ))}
</Container>
```

### Polymorphism

The `Text` component can render as a `span`, `p` or `div` HTML element.

```jsx live
<Container pl={3}>
  <Text as="p">With GitHub Copilot, you’re always in charge.</Text>
</Container>
```

## Component props

### Text

| Name   | Type                                                 |                    Default                     | Description                         |
| :----- | :--------------------------------------------------- | :--------------------------------------------: | :---------------------------------- |
| `as`   | <PropTableValues values={TextTags} addLineBreaks />  | <PropTableValues values={[defaultTextTag]} />  | Applies the underlying HTML element |
| `size` | <PropTableValues values={TextSizes} addLineBreaks /> | <PropTableValues values={[defaultTextSize]} /> | Specify the text size               |
