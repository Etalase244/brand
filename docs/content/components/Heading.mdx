import {
  HeadingSizes,
  HeadingTags,
  defaultHeadingTag,
  defaultHeadingSize,
} from '@primer/react-brand'
import {Box as Container} from '@primer/react'
import {PropTableValues} from '../../src/components'

```js
import {Heading} from '@primer/react-brand'
```

## Examples

### Default

```jsx live
<Container pl={3}>
  <Heading>This is my super sweet heading</Heading>
</Container>
```

### Scale

```jsx live
<Container pl={3}>
  {HeadingSizes.map((size) => (
    <Heading key={size} size={size} as="p">
      Heading {size}
    </Heading>
  ))}
</Container>
```

### Polymorphism

The `Heading` component can render as a `h1`, `h2`, `h3`, `h4`, `h5` or `h6` HTML element.

```jsx live
<Container pl={3}>
  <Heading as="h3">This is my super sweet heading</Heading>
</Container>
```

## Component props

### Heading

| Name   | Type                                                    |                      Default                      | Description                         |
| :----- | :------------------------------------------------------ | :-----------------------------------------------: | :---------------------------------- |
| `as`   | <PropTableValues values={HeadingTags} addLineBreaks />  | <PropTableValues values={[defaultHeadingTag]} />  | Applies the underlying HTML element |
| `size` | <PropTableValues values={HeadingSizes} addLineBreaks /> | <PropTableValues values={[defaultHeadingSize]} /> | Specify the heading size            |
